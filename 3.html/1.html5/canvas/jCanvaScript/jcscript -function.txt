9.动画队列
jc('#myCircle_1').queue(
    function(){
	this.animate({radius:40},5000);
    },
    function(){
	this.animate({x:150},5000,function(){
	    this.color('#ff0000');
	});
	this.animate({y:150},10000,function(){
	    this.color('#000000');
	});
	this.fadeTo(0.5,6500);
    },
    function(){
	this.fadeIn(2000);
    }
);
